const API_URL = '/api/db'; let appData = null; // … (loadData same as public) async function saveData() { const password = prompt('Enter admin password:'); if (!password) return; const res = await fetch(API_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ password, data: appData }) }); const result = await res.json(); if (result.success) { alert('Saved successfully!'); } else { alert('Error: ' + (result.error || 'Unauthorized')); } }
